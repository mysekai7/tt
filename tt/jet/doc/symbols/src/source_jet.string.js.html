 81 	 * 多行或单行字符串模板处理
 82 	 * 
 83 	 * @method template
 84 	 * @memberOf string
 85 	 * 
 86 	 * @param {String} str 模板字符串
 87 	 * @param {Object} obj 要套入的数据对象
 88 	 * @return {String} 返回与数据对象合成后的字符串
 89 	 * 
 90 	 * @example
 91 	 * <script type="text/html" id="user_tmpl">
 92 	 *   <% for ( var i = 0; i < users.length; i++ ) { %>
 93 	 *     <li><a href="<%=users[i].url%>"><%=users[i].name%></a></li>
 94 	 *   <% } %>
 95 	 * </script>
 96 	 * 
 97 	 * Jet().$package(function(J){
 98 	 * 	// 用 obj 对象的数据合并到字符串模板中
 99 	 * 	J.template("Hello, {name}!", {
100 	 * 		name:"Kinvix"
101 	 * 	});
102 	 * };

104 	template = function(str, data){
105 		// Figure out if we're getting a template, or if we need to
106 		// load the template - and be sure to cache the result.
107 		var fn = !/\W/.test(str) ?
108 		  cache[str] = cache[str] ||
109 			tmpl(document.getElementById(str).innerHTML) :
110 		  
111 		  // Generate a reusable function that will serve as a template
112 		  // generator (and which will be cached).
113 		  new Function("obj",
114 			"var p=[],print=function(){p.push.apply(p,arguments);};" +
115 			
116 			// Introduce the data as local variables using with(){}
117 			"with(obj){p.push('" +
118 			
119 			// Convert the template into pure JavaScript
120 			str
121 			  .replace(/[\r\t\n]/g, " ")
122 			  .split("<%").join("\t")
123 			  .replace(/((^|%>)[^\t]*)'/g, "$1\r")
124 			  .replace(/\t=(.*?)%>/g, "',$1,'")
125 			  .split("\t").join("');")
126 			  .split("%>").join("p.push('")
127 			  .split("\r").join("\\'")
128 		  + "');}return p.join('');");
129 		
130 		// Provide some basic currying to the user
131 		return data ? fn( data ) : fn;
132 	};

172 	/**
173 	 * 将 uri 的查询字符串参数映射成对象
174 	 * 
175 	 * @method mapQuery
176 	 * @memberOf string
177 	 * 
178 	 * @param {String} uri 要映射的 uri
179 	 * @return {Object} 按照 uri 映射成的对象
180 	 * 
181 	 * @example
182 	 * Jet().$package(function(J){
183 	 * 	var url = "http://web.qq.com/?qq=4765078&style=blue";
184 	 * 	// queryObj 则得到一个{qq:"4765078", style:"blue"}的对象。
185 	 * 	var queryObj = J.mapQuery(url);
186 	 * };
187 	 */
188 	mapQuery = function(uri){
189 		//window.location.search
190 		var i,
191 			key,
192 			value,
193 			uri = uri || window.location.href,
194 			index = uri.indexOf("?"),
195 			pieces = uri.substring(index + 1).split("&"),
196 			params = {};
197 			
198 		for(i=0; i<pieces.length; i++){
199 			try{
200 				index = pieces[i].indexOf("=");
201 				key = pieces[i].substring(0,index);
202 				value = pieces[i].substring(index+1);
203 				if(!(params[key] = unescape(value))){
204 					throw new Error("uri has wrong query string.");
205 				}
206 			}
207 			catch(e){
208 				//J.out("错误：[" + e.name + "] "+e.message+", " + e.fileName+", 行号:"+e.lineNumber+"; stack:"+typeof e.stack, 2);
209 			}
210 		}
211 		return params;
212 	};

408 	/**
409 	 * 脱去script标签
410 	 * 
411 	 * @memberOf string
412 	 * @return {String} 返回转换后的字符串
413 	 */
414 	stripScripts = function(string, option){
415 		var scripts = '';
416 		var text = string.replace(/<script[^>]*>([\s\S]*?)<\/script>/gi, function(){
417 			scripts += arguments[1] + '\n';
418 			return '';
419 		});
420 		if (option === true){
421 			$exec(scripts);
422 		}else if($type(option) == 'function'){
423 			option(scripts, text);
424 		}
425 		return text;
426 	};

459 	 * @memberOf string
460 	 * @return {String} 返回转换后的字符串

462 	substitute = function(string, object, regexp){
463 		return string.replace(regexp || (/\\?\{([^{}]+)\}/g), function(match, name){
464 			if (match.charAt(0) == '\\') return match.slice(1);
465 			return (object[name] != undefined) ? object[name] : '';
466 		});
467 	};

470 	 * 全局替换指定的字符串
471 	 * 
472 	 * @memberOf string
473 	 * @return {String} 返回替换后的字符串
474 	 */
475 	replaceAll = function(string, reallyDo, replaceWith, ignoreCase) {
476 	    if (!RegExp.prototype.isPrototypeOf(reallyDo)) {
477 	        return string.replace(new RegExp(reallyDo, (ignoreCase ? "gi": "g")), replaceWith);
478 	    } else {
479 	        return string.replace(reallyDo, replaceWith);
480 	    }
481 	};
