 60 	 * @link http://developer.mozilla.org/en/docs/Core_JavaScript_1.5_Reference:Objects:Array:indexOf
 61 	 * @memberOf array

 63 	 * 
 64 	 * @param {Array} arr 要执行操作的数组
 65 	 * @param {Object} obj 要查找的数组的元素
 66 	 * @param {Number} fromIndex 开始的索引编号
 67 	 * 
 68 	 * @return {Number}返回正向查找的索引编号

 70 	indexOf = Array.prototype.indexOf 
 71 		? function(){
 72 			var args = Array.prototype.slice.call(arguments, 1);
 73 			return Array.prototype.indexOf.apply(arguments[0], args);
 74 		}
 75 		: function (arr, obj, fromIndex) {
 76 	
 77 	        if (fromIndex == null) {
 78 	            fromIndex = 0;
 79 	        } else if (fromIndex < 0) {
 80 	            fromIndex = Math.max(0, arr.length + fromIndex);
 81 	        }
 82 	        for (var i = fromIndex; i < arr.length; i++) {
 83 	            if (arr[i] === obj){
 84 	                return i;
 85 	            }
 86 	        }
 87 	        return -1;
 88 	    };

305 	/**
306 	 * 对该数组的每项和前一次调用的结果运行一个函数，收集最后的结果。
307 	 * 
308 	 * @link http://developer.mozilla.org/en/docs/Core_JavaScript_1.8_Reference:Objects:Array:reduce
309 	 * @memberOf array
310 	 * @function
311 	 * 
312 	 * @param {Array} arr 要执行操作的数组
313 	 * @param {Function} fun 要执行的函数
314 	 * @param {Object} contextObj 执行函数时的上下文对象，可以省略
315 	 * 
316 	 * @return {Boolean}
317 	 */
318 	reduce = Array.prototype.reduce 
319 		? function(){
320 			var args = Array.prototype.slice.call(arguments, 1);
321 			return Array.prototype.reduce.apply(arguments[0], args);
322 		}
323 		: function(arr, fun /*, initial*/){
324 			var len = arr.length >>> 0;
325 			if (typeof fun != "function"){
326 				throw new TypeError();
327 			}
328 			// no value to return if no initial value and an empty array
329 			if (len == 0 && arguments.length == 2){
330 				throw new TypeError();
331 			}
332 			var i = 0;
333 			if (arguments.length >= 3){
334 				var rv = arguments[2];
335 			}
336 			else{
337 				do{
338 				    if (i in arr){
339 				      rv = arr[i++];
340 				      break;
341 				    }
342 				
343 				    // if array contains no values, no initial value to return
344 				    if (++i >= len){
345 				    	throw new TypeError();
346 				    }
347 				}
348 				while (true);
349 			}
350 			
351 			for (; i < len; i++){
352 				if (i in arr){
353 					rv = fun.call(null, rv, arr[i], i, arr);
354 				}
355 			}
356 			
357 			return rv;
358 		};

404 				}
405 				while(true);
406 			}

411 				}
412 			}

476 	// 冒泡排序,默认从小到大排序
477 	bubbleSort = function(arr, compareFunc) {
478 		compareFunc = compareFunc || function(num1, num2){
479 			return num1 - num2;
480 		};
481 		//数组长度
482 		var n = arr.length;
483 		//交换顺序的临时变量
484 		var temp;//
485 		//交换标志
486 		var exchange;
487 		//最多做n-1趟排序
488 		for (var time=0; time<n-1; time++){
489 			exchange = false;
490 			for (var i=n-1; i>time; i--) {
491 				if (compareFunc(arr[i], arr[i - 1]) < 0) {
492 				//if (arr[i] < arr[i - 1]) {
493 					exchange = true;
494 					temp = arr[i - 1];
495 					arr[i - 1] = arr[i];
496 					arr[i] = temp;
497 				}
498 			}
499 			//若本趟排序未发生交换，提前终止算法
500 			if (!exchange) {
501 				break;
502 			}
503 		}
504 		return arr;
505 	};

507 	// 二叉搜索
508 	binarySearch = function(arr, item, compareFunc){
509 	    var start = 0;
510 	    var end = arr.length;
511 	    var current = Math.floor(arr.length/2);
512 	    while(end != current){
513 	        if(compareFunc(item, arr[current]) > 0){
514 	            start = current + 1;
515 	        }
516 	        else{
517 	            end = current;
518 	        };
519 	
520 	        current = Math.floor((start + end) / 2);
521 	    };
522 	    return current;
523 	};
