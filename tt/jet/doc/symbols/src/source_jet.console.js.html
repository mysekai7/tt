503 		/**
504 		 * 是否debug环境
505 		 * 
506 		 * @return {Boolean} 是否呢

512 		/**
513 		 * log记录器
514 		 * 
515 		 * @ignore
516 		 * @param {String} msg 信息记录器
517 		 */
518 		function log(msg, type) {
519 			var info;
520 			if (isDebugMode()) {
521 				info = msg + '\n=STACK=\n' + stack();
522 			} else {
523 				if (type == 'error') {
524 					info = msg;
525 				} else if (type == 'warn') {
526 					// TBD
527 				}
528 			}
529 			J.Debug.errorLogs.push(info);
530 		}
531 	
532 		/**
533 		 * 警告信息记录
534 		 * 
535 		 * @param {String} sf 信息模式
536 		 * @param {All} args 填充参数
537 		 */
538 		function warn(sf, args) {
539 			log(write.apply(null, arguments), 'warn');
540 		}
541 	
542 		/**
543 		 * 错误信息记录
544 		 * 
545 		 * @param {String} sf 信息模式
546 		 * @param {All} args 填充参数
547 		 */
548 		function error(sf, args) {
549 			log(write.apply(null, arguments), 'error');
550 		}
551 	
552 		/**
553 		 * 获取当前的运行堆栈信息
554 		 * 
555 		 * @param {Error} e 可选，当时的异常对象
556 		 * @param {Arguments} a 可选，当时的参数表
557 		 * @return {String} 堆栈信息
558 		 */
559 		function stack(e, a) {
560 			function genTrace(ee, aa) {
561 				if (ee.stack) {
562 					return ee.stack;
563 				} else if (ee.message.indexOf("\nBacktrace:\n") >= 0) {
564 					var cnt = 0;
565 					return ee.message.split("\nBacktrace:\n")[1].replace(/\s*\n\s*/g, function() {
566 						cnt++;
567 						return (cnt % 2 == 0) ? "\n" : " @ ";
568 					});
569 				} else {
570 					var entry = (aa.callee == stack) ? aa.callee.caller : aa.callee;
571 					var eas = entry.arguments;
572 					var r = [];
573 					for (var i = 0, len = eas.length; i < len; i++) {
574 						r.push((typeof eas[i] == 'undefined') ? ("<u>") : ((eas[i] === null) ? ("<n>") : (eas[i])));
575 					}
576 					var fnp = /function\s+([^\s\(]+)\(/;
577 					var fname = fnp.test(entry.toString()) ? (fnp.exec(entry.toString())[1]) : ("<ANON>");
578 					return (fname + "(" + r.join() + ");").replace(/\n/g, "");
579 				}
580 			}
581 	
582 			var res;
583 	
584 			if ((e instanceof Error) && (typeof arguments == 'object') && (!!arguments.callee)) {
585 				res = genTrace(e, a);
586 			} else {
587 				try {
588 					({}).sds();
589 				} catch (err) {
590 					res = genTrace(err, arguments);
591 				}
592 			}
593 	
594 			return res.replace(/\n/g, " <= ");
595 		}
596 	
597 		return {
598 			/**
599 			 * 获取当前的运行堆栈信息
600 			 * 
601 			 * @param {Error} e 可选，当时的异常对象
602 			 * @param {Arguments} a 可选，当时的参数表
603 			 * @return {String} 堆栈信息
604 			 */
605 			stack : stack,
606 			/**
607 			 * 警告信息记录
608 			 * 
609 			 * @param {String} sf 信息模式
610 			 * @param {All} args 填充参数
611 			 */
612 			warn : warn,
613 			/**
614 			 * 错误信息记录
615 			 * 
616 			 * @param {String} sf 信息模式
617 			 * @param {All} args 填充参数
618 			 */
619 			error : error,
620 			
621 			/**
622 			 * 是否调试模式
623 			 */
624 			isDebugMode : isDebugMode
625 		};
626
